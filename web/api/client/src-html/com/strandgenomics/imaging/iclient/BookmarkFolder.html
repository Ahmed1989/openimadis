001    package com.strandgenomics.imaging.iclient;
002    
003    /**
004     * client side representation of record bookmarks. a bookmark folder represent
005     * hierarchical organization of bookmarked records
006     * BookmarkFolder contains children BookmarkFolder and list of records under the BookmarkFolder 
007     * 
008     * @author Anup Kulkarni
009     */
010    public class BookmarkFolder  extends ImageSpaceObject {
011            
012            /**
013             * server side representation of path of this folder
014             */
015            public final String absolutePath;
016            /**
017             * parent project id
018             */
019            public final String projectName;
020            
021            BookmarkFolder(String parentProject, String absolutePath)
022            { 
023                    this.absolutePath = absolutePath;
024                    this.projectName = parentProject;
025            }
026            
027            /**
028             * returns all the subfolders for this bookmark folder 
029             * @return all the subfolders
030             */
031            public BookmarkFolder[] getSubfolders()
032            {
033                    return getImageSpace().getSubfolders(this);
034            }
035            
036            /**
037             * returns the bookmarked records for this folder
038             * @return the bookmarked records
039             */
040            public long[] getBookmarkedRecords()
041            {
042                    return getImageSpace().getBookmarkedRecords(this);
043            }
044            
045            /**
046             * create new folder under current folder
047             * @param newFolder name of the bookmark folder
048             */
049            public void createBookmarkFolder(String newFolder)
050            {
051                    getImageSpace().createBookmarkFolder(this, newFolder);
052            }
053            
054            /**
055             * adds bookmark record
056             * @param guid to be bookmarked
057             */
058            public void addBookmark(long guid)
059            {
060                    getImageSpace().addBookmark(this, guid);
061            }
062            
063            @Override
064            public void dispose()
065            { }
066    }
