020    package com.strandgenomics.imaging.iclient;
021    
022    import java.io.File;
023    import java.util.List;
024    import java.util.Map;
025    import java.util.Set;
026    
027    import com.strandgenomics.imaging.icore.app.Parameter;
028    import com.strandgenomics.imaging.icore.app.Priority;
029    import com.strandgenomics.imaging.icore.app.State;
030    
031    /**
032     * Service to enable workflow - executing of task/pipeline by the server
033     * @author arunabha
034     *
035     */
036    public interface ImageSpaceWorkflow {
037    
038            /**
039             * Returns the publisher of the specified application
040             * @param application the application 
041             * @return the publisher for the specified application
042             */
043            public Publisher getPublisher(Application application);
044    
045            /**
046             * Executes the specified registered-application with the specified parameters (if any) 
047             * on the specified list of records as input
048             * @param application the application pre-registered with the server (through plugins to compute-server)
049             * @param parameters the parameters to run the application
050             * @param guids the input records
051             * @param priority priority of the task (running application) under consideration 
052             * @return an handle to the remotely executing task
053             */
054            public Job executeApplication(Application application, Map<String, Object> parameters, Priority priority, String projectName, long ... guids);
055    
056            /**
057             * Schedules the specified registered-application for execution with the specified parameters (if any) 
058             * on the specified list of records as input
059             * @param application the application pre-registered with the server (through plugins to compute-server)
060             * @param parameters the parameters to run the application
061             * @param records the input records
062             * @param priority priority of the task (running application) under consideration 
063             * @param scheduleTime time in milliseconds 
064             * @return an handle to the remotely executing task
065             */
066            public Job scheduleApplication(Application application, Map<String, Object> parameters, Priority priority, String projectName, long scheduleTime, long ... guids);
067    
068            /**
069             * Returns the list of available application with the given category and from the specified publisher 
070             * @param publisher publisher (null will mean a listing across all publisher)
071             * @param categoryName name of the category  (null will mean a listing across all categories)
072             * @return list of available applications
073             */
074            public List<Application> listApplications(Publisher publisher, String categoryName);
075    
076            /**
077             * Returns the required application specific parameters for the specified application 
078             * @param application application under consideration
079             * @return a list of relevant parameters or null, if nothing there
080             */
081            public Set<Parameter> getParameters(Application application) ;
082    
083            /**
084             * Returns the state of the task submitted for execution
085             * @param job handle to the remotely executing task
086             * @return the status code. one of SCHEDULED,WAITING,PAUSED,ALLOCATED,EXECUTING,DELETING, DELETED,ERROR,SUCCESS,TERMINATED
087             */
088            public State getJobState(Job job) ;
089    
090            /**
091             * Reschedule the specified job  which is still to be allocated (to a worker for execution)
092             * @param job handle to the remotely executing task
093             * @param rescheduleTime the new wait time interval
094             * @return true if successful, false otherwise
095             */
096            public boolean rescheduleTask(Job job, long rescheduleTime);
097    
098            /**
099             * Pauses the specified job  which is still to be allocated (to a worker for execution)
100             * @param job handle to the remotely executing task
101             * @return true if successful, false otherwise
102             */
103            public boolean pauseTask(Job job);
104    
105            /**
106             * Resumes the specified job  which is in a paused state
107             * @param job handle to the remotely executing task
108             * @return true if successful, false otherwise
109             */
110            public boolean resumeTask(Job job);
111            
112            /**
113             * Removes the specified job  which is still to be allocated (to a worker for execution)
114             * @param job handle to the remotely executing task
115             * @return true if successful, false otherwise
116             */
117            public boolean removeTask(Job job);
118    
119            /**
120             * Terminate or kill the specified job which is being executed (by an worker)
121             * @param job handle to the remotely executing task
122             * @return true if successful, false otherwise
123             */
124            public boolean terminateTask(Job job);
125    
126            /**
127             * Returns the progress measure (0-100) associated the specified running task
128             * @param job handle to the remotely executing task
129             * @return the progress (0-100)
130             */
131            public int getTaskProgress(Job job);
132            
133            /**
134             * sets the progress measure (0-100) associated the specified running task
135             * @param jobID handle to the remotely executing task
136             * @param progress the progress (0-100)
137             */
138            public void setTaskProgress(long jobID, int progress);
139    
140            /**
141             * Returns the List of parameters used while executing the specified Job
142             * @param job handle to the remotely executing task
143             * @return the list of parameters used when the task was submitted
144             */
145            public Map<String, Object> getTaskParameters(Job job);
146    
147            /**
148             * Returns the list of Record GUIDs which are used as inputs for the specified task
149             * @param job handle to the remotely executing task
150             * @return list of input Records (their GUID)
151             */
152            public long[] getTaskInputs(Job job);
153            
154            /**
155             * Returns the list of Record GUIDs which are potentially created by this Job
156             * Note that is call is valid only after the said job is ended successfully
157             * @param job handle to the remotely executing task
158             * @return list of Records (their GUID) that are create or null if there are none
159             */
160            public long[] getTaskOutputs(String accessToken, long jobID);
161            
162            /**
163             * upload the task log file for the specified job
164             * @param jobID specified job id
165             * @param logFile log file to be attached with the log
166             */
167            public void uploadTaskLog(long jobID, File logFile);
168    }
