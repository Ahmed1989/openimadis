020    package com.strandgenomics.imaging.iclient;
021    
022    import com.strandgenomics.imaging.icore.Rank;
023    
024    /**
025     * Administrative APIs - to manager projects and users
026     * @author arunabha
027     *
028     */
029    public interface ImageSpaceManagement {
030            
031            /**
032             * Creates a new project with the specified name and notes. 
033             * Note that the login user must be a TeamLeader and above to create a new project
034             * Note that by default, the creator is assigned as the project manager
035             * @param projectName name of the project (must be unique across all projects)
036             * @param notes project description
037             * @param quota quota in GB
038             * @return newly created project
039             */
040            public Project createNewProject(String projectName, String notes, double quota);
041            
042            /**
043             * Note that the login user needs to an Administrator for this API will be successfully 
044             * Allows a new user with the specified login and other details to access the system with
045             * external authentication - say LDAP
046             * @param login login name of the user, must be unique
047             * @param email email of the user
048             * @param rank rank of the user
049             * @return true is successful, false otherwise
050             */
051            public boolean allowExternalUser(String login, String email, Rank rank);
052            
053            /**
054             * Note that the login user needs to an Administrator for this API will be successfully 
055             * Creates a new user with the specified login and other details
056             * @param login login name of the user, must be unique
057             * @param password password of the user
058             * @param email email of the user
059             * @param fullName full name of the user
060             * @param rank rank of the user
061             * @return true is successful, false otherwise
062             */
063            public boolean createInternalUser(String login, String password, String email, String fullName, Rank rank);
064            
065            /**
066             * Deletes the specified project
067             * @param project the project to delete
068             * @return the handle to the long running task
069             */
070            public Task deleteProject(String projectName);
071            
072            /**
073             * Archives the specified project
074             * @param project the project to delete
075             * @return the handle to the long running task
076             */
077            public Task archiveProject(String projectName);
078    
079            /**
080             * Restore the specified archived project
081             * @param project the project to delete
082             * @return the handle to the long running task
083             */
084            public Task restoreProject(String projectName);
085    }
