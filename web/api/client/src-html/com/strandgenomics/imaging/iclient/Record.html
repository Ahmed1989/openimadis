166             * @param parent owning project
167             * @param uploadedBy user who had uploaded this record
168             * @param uploadTime time of upload
169             * @param isWritable checks whether the current login user has write permission
170             * @param signature unique signature associated with this record
171             * @param macAddress MAC address of the acquisition machine
172             * @param ipAddress IP address of the acquisition machine
173             * @param creationTime acquisition time
174             * @param seriesNumber the series number of this record if any, otherwise zero
175             * @param noOfFrames number of frames within this record
176             * @param noOfSlices number of slices (z) within this record
177             * @param noOfChannels number of channels within this record
178             * @param noOfSites number of sites within this record
179             * @param imageWidth width of a pixel-data (image) within this record
180             * @param imageHeight height of a pixel-data (image) within this record
181             * @param pixelDepth depth of a pixel in the pixel-data
182             * @param pixelSizeX x-size (in micron) of the pixel
183             * @param pixelSizeY y-size (in micron) of the pixel
184             * @param imageType type of the image, by default it is Gray-scale
185             * @param sourceFormat file format of the original source files
186             */

678            @Override
679            public synchronized List<BufferedImage> getChannelOverlayImagesForSlices(int frameNo, int siteNo, int imageWidth, 
680                            boolean useChannelColor) throws IOException
681            {
682                    //makes a system call to get it done
683                    InputStream iStream = getImageSpace().getChannelOverlaidImagesForSlices(this.getGUID(), 
684                                    frameNo, siteNo, imageWidth, useChannelColor);
685                    
686                    File gzTarball = File.createTempFile(frameNo+"_slice_"+siteNo, ".tar.gz", Constants.TEMP_DIR);
687                    gzTarball.deleteOnExit();
688                    
689                    BufferedInputStream input = null;
690                    BufferedOutputStream output = null;
691                    
692                    File destFolder = null;
693                    File[] sliceFiles = null;
694                    List<BufferedImage> images = null;
695                                    
696                    try
697                    {
698                            input = new BufferedInputStream(iStream);
699                            output = new BufferedOutputStream(new FileOutputStream(gzTarball));
700                            long dataLength = Util.transferData(input, output);
701                            
702                            Util.closeStreams(input, output);
703                            
704                            System.out.println("successfully downloaded "+dataLength);
705                            
706                            destFolder = new File( Constants.TEMP_DIR, "slices_"+System.currentTimeMillis());
707                            destFolder.mkdir();
708                            
709                            Archiver.unTar(destFolder, gzTarball, true);
710                            sliceFiles = destFolder.listFiles();
711                            
712                            images = readBufferedImages(sliceFiles);
713                    }
714                    finally
715                    {
716                            gzTarball.delete();
717                            for(File f : sliceFiles)
718                            {
719                                    if(f != null) f.delete();
720                            }
721                            destFolder.delete();
722                    }
723                    
724                    return images;
725            }
726    
727            private List<BufferedImage> readBufferedImages(File[] sliceFiles) throws IOException 
728            {
729                    List<BufferedImage> images = new ArrayList<BufferedImage>();
730                    for(File f : sliceFiles)
731                    {
732                            images.add( ImageIO.read(f) );
733                    }
734                    
735                    return images;
736            }
