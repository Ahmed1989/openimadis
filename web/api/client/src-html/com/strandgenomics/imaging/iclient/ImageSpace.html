021    package com.strandgenomics.imaging.iclient;
022    
023    import java.rmi.RemoteException;
024    import java.util.List;
025    import java.util.Set;
026    
027    import com.strandgenomics.imaging.icore.SearchCondition;
028    
029    /**
030     * The main class representing the Enterprise IMG Server at the client side
031     * @author arunabha
032     */
033    public interface ImageSpace {
034            
035            /**
036         * returns the host address (of Enterprise IMG Server) connected to, 
037         * null otherwise
038         */
039        public String getHost();
040    
041        /**
042         * @return the server port number connected to, -1 otherwise
043         */
044        public int getPort();
045    
046        /**
047         * Enterprise IMG Server supports authentication.
048         * @returns the logged user name
049         */
050        public String getUser();
051    
052            /**
053             * returns the access key associated with active user session. the access
054             * key is generated by server for given authcode, user and client id. access
055             * key is unique for every login. this api fascilitates transfer of user
056             * session.
057             * 
058             * @return access key
059             */
060            public String getAccessKey();
061            
062            /**
063             * this call is equivalent to login call but with the given user session 
064             * @param useSSL true if ssl is used
065             * @param host host name of the server
066             * @param port port name of the server
067             * @param accessKey associated with the session
068             * @return true if successful, false otherwise
069             */
070            public boolean setAccessKey(boolean useSSL, String host, int port, String accessKey);
071    
072        /**
073         * Logs in the user to Enterprise IMG Server
074         * @param host the IP address of the Enterprise IMG Server machine
075         * @param port the port to use
076         * @param appID Client/Application Identifier - obtained after registering the client/Application with iManage (web client) 
077         * @param authCode the authorization grant obtained by the user for the above mentioned clientID (again using iManage)
078         * @throws AuthenticationException
079         */
080        public boolean login(boolean useSSL, String host, int port, String appID, String authCode) 
081                    throws AuthenticationException;
082        
083        /**
084         * Logs out from the iManager server by surrending the underlying acess-token
085         */
086        public void logout();
087        
088        /**
089         * The list of active projects that the connected user have permission to read
090         * @return list of active projects that the connected user have permission to read
091         */
092        public List<Project> getActiveProjects();
093        
094        /**
095         * The list of archived projects that the connected user have permission to browse, may be null otherwise
096         * @return list of active projects that the connected user have permission to read
097         */
098        public List<Project> getArchivedProjects();
099        
100        /**
101         * find the specified project
102         * @param projectName project name
103         * @return project 
104         */
105        public Project findProject(String projectName);
106        
107        /**
108         * Search AVADIS IMG Server for records matching the specified search text
109         * @param freeText the search text
110         * @param projectNames list of projects where the search is restricted (can be null)
111         * @param filters list of search conditions
112         * @param maxResult maximum number of result to fetch
113         * @return a list of records (their GUIDs)
114         * @throws RemoteException
115         */
116        public long[] search(String freeText, Set<String> projectNames, Set<SearchCondition> filters, int maxResult);
117        
118        
119            /**
120             * Find the record for the specified global universal identifier for record
121             * @param guid the universal identifier for records
122             * @return the record record for the specified id
123             */
124            public Record findRecordForGUID(long guid);
125            
126            /**
127             * Find the record for the specified global universal identifier for record
128             * @param guid the universal identifier for records
129             * @return the record
130             */
131            public Record[] findRecordForGUIDs(long[] guid);
132    
133        /**
134         * list all the available acquisition profiles
135         * @return available acquisition profiles
136         */
137        public abstract List<AcquisitionProfile> listAcquisitionProfiles();
138        
139        /**
140         * returns name of the microscope attached to current machine
141         * @return name of the microscope attached to current machine
142         */
143        public abstract String getMicroscope();
144        
145        /**
146         * request acquisition license for current machine and user
147         * @param user specified user
148         * @return true if license is granted, false otherwise
149         */
150        public abstract boolean requestAcquisitionLicense(String user);
151        
152        /**
153         * surrender acquisition license from current machine/user
154         */
155        public abstract void surrenderAcquisitionLicense();
156    }
