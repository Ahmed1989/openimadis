001    package com.strandgenomics.imaging.iclient.daemon;
002    
003    import java.io.Serializable;
004    
005    /**
006     * The details of the user session
007     * 
008     * @author Anup Kulkarni
009     */
010    public class UserSessionSpecification implements Serializable{
011            /**
012             * hostname of server
013             */
014            public final String host;
015            /**
016             * port of the server
017             */
018            public final int port;
019            /**
020             * protocol
021             */
022            public final boolean useSSL;
023            /**
024             * access key 
025             */
026            public final String accessKey;
027            /**
028             * logged in user
029             */
030            public final String userLogin;
031            /**
032             * name of the project to which user is logged in
033             */
034            public final String projectName;
035            
036            public UserSessionSpecification(String userLogin, String accessKey, String projectName, String host, int port, boolean useSSL)
037            {
038                    this.userLogin = userLogin;
039                    this.accessKey = accessKey;
040                    this.projectName = projectName;
041                    this.host = host;
042                    this.port = port;
043                    this.useSSL = useSSL;
044            }
045    }
