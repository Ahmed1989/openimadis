001    package com.strandgenomics.imaging.iclient.daemon;
002    
003    import java.rmi.Remote;
004    import java.rmi.RemoteException;
005    import java.util.List;
006    
007    /**
008     * apis required for record upload service 
009     * 
010     * @author Anup Kulkarni
011     */
012    public interface UploadDaemonService extends Remote{
013            /**
014             * submits upload request
015             * @param spec
016             * @throws RemoteException
017             * @throws Exception 
018             */
019            public void submitUploadRequest(UploadSpecification spec) throws RemoteException, Exception;
020            /**
021             * get the status of uploads
022             * @param userName name of user
023             * @param projectName name of target project
024             * @throws RemoteException
025             * @throws Exception
026             */
027            public List<UploadSpecification> getStatus(String userName, String projectName) throws RemoteException, Exception;
028            /**
029             * clears the completed(either failed or successful) uploads from the daemon session
030             * @param user name of user
031             * @param projectName name of target project
032             * @throws RemoteException
033             * @throws Exception
034             */
035            public void clearCompleted(String user, String projectName) throws RemoteException, Exception;
036            /**
037             * cancels the selected uploads
038             * @param userName
039             * @param uploadIds
040             * @throws RemoteException
041             * @throws Exception
042             */
043            public void cancelSelected(String userName, List<Long>uploadIds) throws RemoteException, Exception;
044    }
