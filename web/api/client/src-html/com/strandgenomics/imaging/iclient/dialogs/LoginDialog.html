082        public LoginDialog(JFrame applicationFrame, String name, boolean maskLoginField) 
083        {
084            super(applicationFrame, name, true);
085            this.maskLoginField  = maskLoginField;

094        public void readyDialog()
095        {
096            int preferredWidth  = 320;
097            int preferredHeight = 280;
098            
099            setPreferredSize(new Dimension(preferredWidth, preferredHeight));
100            
101            Point location = getMiddlePosition(preferredWidth, preferredHeight);
102            setLocation(location.x, location.y);
103    
104            addWindowListener(new WindowAdapter() 
105            {
106                    public void windowClosing(WindowEvent e) 
107                    {
108                            setVisible(false);
109                    }
110            });
111    
112            setResizable(true);
113            pack();
114        }

116        /**
117         * validate and set connection parameters 
118         * @return true if done successfully, false otherwise
119         */
120            private boolean validateAndSetParameters()
121            {
122                    try
123                    {
124                            docModel.validateAndSetParameters();
125                            return true;
126                    }
127                    catch(ConfigurationException cax)
128                    {
129                            showErrorMessage(cax.getCategory(), cax.getMessage() +(cax.getValue() == null ? "" : cax.getValue()));
130                            return false;
131                    }
132        }

134            /**
135             * Creates and return the JPanel representing the main GUI of this dialog
136             * @return The JPanel of the UI
137             */
138            private JPanel createUI()
139            {
140            JPanel contentPane = new JPanel();
141            contentPane.setLayout( new BorderLayout() );
142            
143            contentPane.add( createTabPanel(), BorderLayout.CENTER );
144            contentPane.add( createButtonPanel(), BorderLayout.SOUTH );
145            
146            return contentPane;
147            }
148            
149            /**
150             * Creates the Button Panel and return the JPanel representing the UI
151             * @return the JPanel representing the Button Panel
152             */
153            private JPanel createButtonPanel()
154            {
155            @SuppressWarnings("serial")
156                    AbstractAction okAction = new AbstractAction("OK", null)
157            {
158                public void actionPerformed(ActionEvent e){
159    
160                    boolean validParameters = validateAndSetParameters();
161    
162                    if(!validParameters)
163                    {
164                        return;
165                    }
166    
167                    cancelled = false;
168                    setVisible(false);
169                    dispose();
170                }
171            };
172            
173            @SuppressWarnings("serial")
174                    AbstractAction cancelAction = new AbstractAction("Cancel", null)
175            {
176                public void actionPerformed(ActionEvent e)
177                {
178                    cancelled = true;
179                    setVisible(false);
180                    dispose();
181                }
182            };
183            
184            //associate keyboard shortcuts for OK and Cancel
185            KeyStroke enterKeyStroke = KeyStroke.getKeyStroke(KeyEvent.VK_ENTER, 0, false);
186            getRootPane().getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW).put(enterKeyStroke, "ENTER");
187            getRootPane().getActionMap().put("ENTER", okAction);
188    
189            KeyStroke escapeKeyStroke = KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0, false);
190            getRootPane().getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW).put(escapeKeyStroke, "ESCAPE");
191            getRootPane().getActionMap().put("ESCAPE", cancelAction);
192            
193            Action[] buttonActions = {okAction, cancelAction};
194            return new HorizontalButtonPanel(buttonActions, FlowLayout.RIGHT);
195            }

198            /**
199             * Creates and return the JPanel representing the tabbed-panel in the UI
200             * @return return the JPanel representing the tabbed-panel in the UI
201             */
202        private JTabbedPane createTabPanel()
203        {
204            //the components are final since they are used within the anonymous class
205            final JTextField proxyHostBox = new JTextField(docModel.getProxyServerAddressField(), null, 20);
206            proxyHostBox.setEnabled(docModel.useProxy());  
207    
208            final JTextField proxyPortBox = new JTextField(docModel.getProxyPortField(), null, 20);
209            proxyPortBox.setEnabled(docModel.useProxy());
210           
211            final JTextField proxyUserBox = new JTextField(docModel.getProxyUserField(), null, 20);
212            proxyUserBox.setEnabled(docModel.useProxy());
213         
214            final JPasswordField proxyPasswordBox = new JPasswordField(docModel.getProxyPasswordField(), null, 20);
215            proxyPasswordBox.setEnabled(docModel.useProxy());
216            
217            final JCheckBox proxyEnableBox = new JCheckBox();
218            proxyEnableBox.setFont(LABEL_FONT);
219            proxyEnableBox.setSelected(docModel.useProxy());
220    
221            @SuppressWarnings("serial")
222                    AbstractAction checkProxyAction = new AbstractAction("", null)
223            {
224                public void actionPerformed(ActionEvent e)
225                {
226                    boolean useProxy = proxyEnableBox.isSelected();;
227                    proxyHostBox.setEnabled(useProxy);
228                    proxyPortBox.setEnabled(useProxy);
229                    proxyUserBox.setEnabled(useProxy);
230                    proxyPasswordBox.setEnabled(useProxy);
231                    
232                    docModel.setProxyEnabled(useProxy);
233                }
234            };
235            proxyEnableBox.setAction(checkProxyAction);
236            
237            JPasswordField passwordBox = new JPasswordField(docModel.getPasswordField(), null, 20);
238            if(maskLoginField)
239                    passwordBox.setText("");
240            JTextField hostIPBox   = new JTextField(docModel.getServerAddressField(), null, 20);
241            hostIPBox.setEnabled(!maskLoginField);
242    
243            JTextField hostPortBox = new JTextField(docModel.getServerPortField(), null, 20);
244            hostPortBox.setEnabled(!maskLoginField);
245            JTextField loginBox    = new JTextField(docModel.getLoginField(), null, 20);
246            loginBox.setEnabled(!maskLoginField);
247    
248            JTabbedPane tabPane = new JTabbedPane();
249            tabPane.addTab("Primary", constructPrimaryPanel(hostIPBox,hostPortBox,loginBox,passwordBox));
250            tabPane.addTab("Proxy",   constructProxyPanel(proxyHostBox, proxyPortBox, proxyUserBox, proxyPasswordBox, proxyEnableBox));
251    
252            tabPane.setBorder(BorderFactory.createEmptyBorder(5,5,5,5));
253            return tabPane;
254        }

256        /**
257         * Creates and returns JPanel representing the primary connection panel
258         * @param hostIPBox the text field for the server address
259         * @param hostPortBox the text field for the server port
260         * @param loginBox the text field for the login

272            preferencePanel.add(ipLabel);
273            preferencePanel.add(hostIPBox);
274    
275            JLabel portLabel = new JLabel("Server Port:");
276            portLabel.setFont(LABEL_FONT);
277            preferencePanel.add(portLabel);
278            preferencePanel.add(hostPortBox);
279    
280            JLabel loginLabel = new JLabel("Login Name:");
281            loginLabel.setFont(LABEL_FONT);
282            preferencePanel.add(loginLabel);
283            preferencePanel.add(loginBox);
284    
285            JLabel passLabel = new JLabel("Password:");
286            passLabel.setFont(LABEL_FONT);
287            preferencePanel.add(passLabel);
288            preferencePanel.add(passwordBox);

291        }

293        /**
294         * Creates and returns JPanel representing the proxy connection panel
295         * @param proxyHostBox the text field for the proxy server address
296         * @param proxyPortBox the text field for the proxy server port
297         * @param proxyUserBox  the text field for the proxy user
298         * @param proxyPasswordBox  the password text field for proxy user's password

303                    JTextField proxyUserBox, JPasswordField proxyPasswordBox, JCheckBox proxyEnableBox)
304        {
305    
306            JPanel proxyPanel = new JPanel(new GridLayout(5,2,10,10));
307            proxyPanel.setBorder(BorderFactory.createEmptyBorder(5,5,5,5));
308    
309            JLabel serverLabel = new JLabel("Proxy Host:");
310            serverLabel.setPreferredSize(LABEL_SIZE);
311            serverLabel.setFont(LABEL_FONT);
312    
313            proxyPanel.add(serverLabel);
314            proxyPanel.add(proxyHostBox);
315    
316            JLabel portLabel = new JLabel("Proxy Port:");
317            portLabel.setPreferredSize(LABEL_SIZE);
318            portLabel.setFont(LABEL_FONT);
319    
320            proxyPanel.add(portLabel);
321            proxyPanel.add(proxyPortBox);
322    
323            JLabel userLabel = new JLabel("User:");
324            userLabel.setPreferredSize(LABEL_SIZE);
325            userLabel.setFont(LABEL_FONT);
326    
327            proxyPanel.add(userLabel);
328            proxyPanel.add(proxyUserBox);
329    
330            JLabel passLabel = new JLabel("Password:");
331            passLabel.setPreferredSize(LABEL_SIZE);
332            passLabel.setFont(LABEL_FONT);
333    
334            proxyPanel.add(passLabel);
335            proxyPanel.add(proxyPasswordBox);
336

338            proxyLabel.setFont(LABEL_FONT);
339    
340            proxyPanel.add(proxyLabel);
341            proxyPanel.add(proxyEnableBox);
342            
343            return proxyPanel;
344        }

364        public static Point getMiddlePosition(int width, int height)
365        {
366            Toolkit toolBox = Toolkit.getDefaultToolkit();
367            //get the actual physical size of the physical screen
368            Dimension screenBounds = toolBox.getScreenSize();
369            
370            if(width >= screenBounds.width || height > screenBounds.height)
371                return new Point(0,0);
372    
373            int x = (screenBounds.width - width)/2;
374            int y = (screenBounds.height - height)/2;
375    
376            return new Point(x,y);
377        }

380        /**
381         * Create the GUI and show it.  For thread safety,
382         * this method should be invoked from the
383         * event-dispatching thread.
384         */
385        private static void createAndShowGUI() {
386    
387            //Create and set up the window.
388            JFrame frame = new JFrame("ArrayAssist Enterprise Editon");
389            JFrame.setDefaultLookAndFeelDecorated(true);
390    
391            try {
392                UIManager.setLookAndFeel("com.sun.java.swing.plaf.windows.WindowsLookAndFeel");
393                SwingUtilities.updateComponentTreeUI(frame);
394            }
395            catch(Exception ex)
396            {}
397            
398            frame.pack();
399            frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
400            frame.setVisible(false);
401    
402            LoginDialog dialog = new LoginDialog(frame, "Server Connection Preferences", false);
403            dialog.setVisible(true);
404        }
