042            /** server port */
043            protected Document serverPort = null;
044            /** server login */
045            protected Document loginName  = null;
046            /** login password */
047            protected Document password   = null;
048            /** proxy server address */
049            protected Document proxyHost  = null;
050            /** proxy port */
051            protected Document proxyPort  = null;
052            /** proxy username */
053            protected Document proxyUser  = null;
054            /** proxy user password */
055            protected Document proxyPassword = null;
056            /** checks whether to use proxy */
057            protected boolean useProxy = false;
058            /** checks whether to use ssl */
059            protected boolean useSSL = false;
060    
061            public LoginDocomentModel() 
062            {
063                    serverIP   = new PlainDocument();
064                    serverPort = new PlainDocument();
065                    loginName  = new PlainDocument();
066                    password   = new PlainDocument();
067                    proxyHost  = new PlainDocument();
068                    proxyPort  = new PlainDocument();
069                    proxyUser  = new PlainDocument();
070                    proxyPassword = new PlainDocument();
071                    //initializes the fields
072                    initialize();
073                    //initializes the document listeners
074                    initializeListeners();
075            }
076            
077            /**
078             * initializes the model with the underlying connection preference object
079             * @see ConnectionPreferences
080             */
081            protected void initialize()
082            {
083                    ConnectionPreferences pref = ConnectionPreferences.getInstance();
084    
085                    useSSL = pref.sslEnabled();
086                    useProxy = pref.toUseProxy();
087    
088                    try
089                    {
090                            if (pref.getHostAddress() != null)
091                                    serverIP.insertString(0, pref.getHostAddress(), null);
092                            if (pref.getHostPort() != null)
093                                    serverPort.insertString(0, pref.getHostPort().toString(), null);
094                            if (pref.getLoginName() != null)
095                                    loginName.insertString(0, pref.getLoginName(), null);
096                            if (pref.getPassword() != null)
097                                    password.insertString(0, pref.getPassword(), null);
098                            if (pref.getProxyAddress() != null)
099                                    proxyHost.insertString(0, pref.getProxyAddress(), null);
100                            if (pref.getProxyPort() != null)
101                                    proxyPort.insertString(0, pref.getProxyPort().toString(), null);
102                            if (pref.getProxyUser() != null)
103                                    proxyUser.insertString(0, pref.getProxyUser(), null);
104                            if (pref.getProxyPassword() != null)
105                                    proxyPassword.insertString(0, pref.getProxyPassword(), null);
106                    } 
107                    catch (BadLocationException bad) //not likely to happen here
108                    {}
109            }
110

129                            }

133                                    // TODO Auto-generated method stub
134                                    
135                            }
136                    });
137            }
138    
139            /**
140             * validate and set the connection preferences. This methods throws ConfigurationException
141             * with relevant messages if certain fields are not populated with relevant information
142             * @see ConnectionPreferences
143             * @throws ConfigurationException if the fields are not populated properly
144             */
145            public void validateAndSetParameters() throws ConfigurationException 
146            {
147                    if (validateIPAddress(getServerAddress()) == null) {
148                            throw new ConfigurationException("Configuration Error",
149                                            "Illegal host internet address ", getServerAddress());
150                    }
151    
152                    if (getServerPort() == null) {
153                            throw new ConfigurationException("Configuration Error",
154                                            "Illegal Server Port ", getFieldValue(serverPort));
155                    }
156    
157                    if (getLogin() == null) {
158                            throw new ConfigurationException("Configuration Error",
159                                            "Login name cannot be empty ", getLogin());
160                    }
161    
162                    if (getPassword() == null) {
163                            throw new ConfigurationException("Configuration Error",
164                                            "Password field cannot be empty ");
165                    }
166    
167                    ConnectionPreferences pref = ConnectionPreferences.getInstance();
168                    pref.setServerSettings(getServerAddress(), getServerPort(), getLogin(),getPassword(), useSSL);
169                    pref.save();
170    
171                    // now set proxy settings in the system properties
172                    if (useProxy) 
173                    {
174                            if (validateIPAddress(getProxyServerAddress()) == null) 
175                            {
176                                    throw new ConfigurationException("Configuration Error",
177                                                    "Illegal Proxy Address", getProxyServerAddress());
178                            }
179    
180                            if (getProxyPort() == null)
181                            {
182                                    throw new ConfigurationException("Configuration Error",
183                                                    "Illegal Proxy Port ", getFieldValue(proxyPort));
184                            }
185                    }
186    
187                    pref.setProxySettings(getProxyServerAddress(), getProxyPort(), getProxyUser(), getProxyPassword(), useProxy);
188                    pref.save();
189            }
190    
191            private String getFieldValue(Document doc) 
192            {
193                    try 
194                    {
195                            return trim(doc.getText(0, doc.getLength()));
196                    } 
197                    catch (BadLocationException e) 
198                    {
199                            return null;
200                    }
201            }

247            private String validateIPAddress(String str)
248            {
249                    String hostAddress = null;
250                    try {
251                            String host = trim(str);
252                            if (host == null)
253                                    return null;
254    
255                            InetAddress ip = InetAddress.getByName(host);
256                            if (ip != null) {
257                                    hostAddress = host;
258                            }
259                    } 
260                    catch (Exception ex) 
261                    {}
262                    return hostAddress;
263            }
264    
265            private String trim(String str)
266            {
267                    if (str == null)
268                            return null;
269    
270                    str = str.trim(); // trim ends
271                    if (str.length() == 0) {
272                            str = null;
273                    }
274    
275                    return str;
276            }
277    
278            private Integer validatePort(String str)
279            {
280                    try
281                    {
282                            String port = trim(str);
283                            if (port == null)
284                                    return null;
285                            return Integer.parseInt(port);
286                    } 
287                    catch (Exception ex)
288                    {}
289                    return null;
290            }
