001    package com.strandgenomics.imaging.iclient.dialogs;
002    
003    import java.awt.BorderLayout;
004    import java.awt.Dimension;
005    import java.awt.Window;
006    
007    import javax.swing.BorderFactory;
008    import javax.swing.JDialog;
009    import javax.swing.JFrame;
010    import javax.swing.JLabel;
011    import javax.swing.JPanel;
012    import javax.swing.JProgressBar;
013    import javax.swing.SwingUtilities;
014    import javax.swing.UIManager;
015    
016    public class ProgressDialog extends JDialog  {
017            
018        private JLabel title;
019        private JProgressBar progressBar;
020        private Window owner;
021            
022            public ProgressDialog(JFrame applicationFrame, String name) {
023                    super(applicationFrame, name, true);
024                    init(owner);
025            }
026            
027            
028            private void init(Window owner) {
029                    this.owner = owner;
030                    JPanel progress = createProgressPanel();
031            setContentPane(progress);
032            setUndecorated(true);
033            this.setPreferredSize(new Dimension(250, 50));
034            pack();
035                    
036            }
037            
038        private JPanel createProgressPanel() {
039            JPanel panel = new JPanel(new BorderLayout());
040            panel.setBorder(BorderFactory.createTitledBorder(""));
041            title = new JLabel("Progress .. ");
042    
043            progressBar = new JProgressBar(0, 100);
044          //  progressBar.setBorder(new BevelBorder(BevelBorder.LOWERED));
045            progressBar.setIndeterminate(true);
046    
047            panel.add(title, BorderLayout.NORTH);
048            panel.add(progressBar, BorderLayout.CENTER);
049    
050            return panel;
051        }
052    
053        public final void pack() {
054            super.pack();
055            setLocationRelativeTo(owner);
056        }
057        
058        public void setMessage(String message){
059            if (message != null)
060                    title.setText(message);
061        }
062        
063        /**
064         * Create the GUI and show it.  For thread safety,
065         * this method should be invoked from the
066         * event-dispatching thread.
067         */
068        private static void createAndShowGUI() {
069    
070            //Create and set up the window.
071            JFrame frame = new JFrame("Acquisition Client");
072            JFrame.setDefaultLookAndFeelDecorated(true);
073    
074            try {
075                UIManager.setLookAndFeel("com.sun.java.swing.plaf.windows.WindowsLookAndFeel");
076                SwingUtilities.updateComponentTreeUI(frame);
077            }
078            catch(Exception ex)
079            {}
080            
081            frame.pack();
082            frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
083            frame.setVisible(false);
084    
085            ProgressDialog dialog = new ProgressDialog(frame, "");
086            dialog.setVisible(true);
087        }
088        
089        public static void main(String[] args) {
090    
091            //Schedule a job for the event-dispatching thread:
092            //creating and showing this application's GUI.
093            javax.swing.SwingUtilities.invokeLater(new Runnable() {
094                public void run() {
095                    try {
096                        createAndShowGUI();
097                    }
098                    catch(Exception ex){
099                        ex.printStackTrace();
100                    }
101                }
102            });
103        }
104    
105    }
