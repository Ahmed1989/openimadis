027                    }
028                    String hostIP = args[0];
029                    int hostPort = Integer.parseInt(args[1]);

032    
033                    ImageSpace ispace = ImageSpaceObject.getConnectionManager();

038                    Ellipse e = new Ellipse(10, 20, 30, 60);

042                    vObjects.add(rect);
043                    
044                    LineSegment l = new LineSegment(50, 70, 10, 20);
045                    vObjects.add(l);
046                    
047                    TextBox tb = new TextBox(0, 0, 10, 20, "test comment");
048                    vObjects.add(tb);
049                    
050            Record r = ImageSpaceObject.getImageSpace().findRecordForGUID(1);
051            r.addVisualObjects(vObjects, "overlay", new VODimension(0,0,0));
