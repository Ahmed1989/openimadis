021    package com.strandgenomics.imaging.iclient.test;
022    
023    import java.io.File;
024    import java.io.FileNotFoundException;
025    import java.io.IOException;
026    
027    import java.util.List;
028    
029    import com.strandgenomics.imaging.iclient.ImageSpaceObject;
030    import com.strandgenomics.imaging.iclient.Project;
031    
032    public class Importer {
033            
034            private String userName;
035            private int port;
036            private String host;
037            private String password;
038            
039            public Importer(String host, int port, String userLogin, String password)
040            {
041                    this.password = password;
042                    this.port = port;
043                    this.host = host;
044                    this.userName = userLogin;
045            }
046            
047            public void login()
048            {
049                    ImageSpaceObject.getImageSpace().login(false, host, port, userName, password);
050            }
051            
052            
053            public Project getProject(String projectName) 
054            {
055                    List<Project> activeProjects = ImageSpaceObject.getConnectionManager().getActiveProjects();
056                    for(Project p : activeProjects)
057                    {
058                            if(p.getName().equals(projectName))
059                            {
060                                    return p;
061                            }
062                    }
063                    return null;
064            }
065    
066            public static void main(String ... args) throws IOException
067            {
068                    if(args == null || args[0].equals("--help") || args[0].equals("?"))
069                    {
070                            System.out.println("Usage:java -Xmx1024m -cp <classpath> com.strandgenomics.imaging.iclient.test.Importer -h<server-ip> -p<server-port> -u<user-login> -P<user-password> -j<project-name> -f<index file or folder with quotes> [-r ( recursive indexing)] [ -m (for auto merge)]" );
071                    }
072                    
073                    String host = null;
074                    int port = 8080;
075                    
076                    String userLogin = null;
077                    String password = null;
078                    
079                    String indexFolder = null;
080                    String projectName = null;
081                    
082                    boolean recursive = false;
083                    boolean autoMerge = false; 
084                    
085                    for(String arg : args)
086                    {
087                            if(arg.startsWith("-h"))
088                            {
089                                    host = arg.substring(2);
090                                    continue;
091                            }
092                            
093                            if(arg.startsWith("-p"))
094                            {
095                                    port = Integer.parseInt(arg.substring(2));
096                                    continue;
097                            }
098                            
099                            if(arg.startsWith("-u"))
100                            {
101                                    userLogin = arg.substring(2);
102                                    continue;
103                            }
104                            
105                            if(arg.startsWith("-P"))
106                            {
107                                    password = arg.substring(2);
108                                    continue;
109                            }
110                            
111                            if(arg.startsWith("-j"))
112                            {
113                                    projectName = arg.substring(2);
114                                    continue;
115                            }
116                            
117                            if(arg.startsWith("-r"))
118                            {
119                                    recursive = true;
120                                    continue;
121                            }
122                            
123                            if(arg.startsWith("-m"))
124                            {
125                                    autoMerge = true;
126                                    continue;
127                            }
128                            
129                            if(arg.startsWith("-f")){
130                                    indexFolder = arg.substring(2);
131                                    continue;
132                            }
133                    }
134                    
135                    System.out.println("host="+host);
136                    System.out.println("port="+port);
137                    
138                    System.out.println("userLogin="+userLogin);
139                    System.out.println("password="+password);
140                    
141                    System.out.println("indexFolder="+indexFolder);
142                    System.out.println("projectName="+projectName);
143                    
144                    System.out.println("recursive="+recursive);
145                    System.out.println("autoMerge="+autoMerge);
146                    
147                    File f = new File(indexFolder).getCanonicalFile();
148                    if(!f.exists()) throw new FileNotFoundException(indexFolder);
149                    
150                    Importer importer = new Importer(host,port,userLogin,password);
151                    importer.login();
152                    try
153                    {
154                            Project p = importer.getProject(projectName);
155                            if(p == null)
156                            {
157                                    System.out.println("Named project ("+projectName +") not found, stopping");
158                                    return;
159                            }
160                            
161    //                      p.createRecord(f, recursive, autoMerge);
162                    }
163                    catch(Exception ex)
164                    {
165                            ex.printStackTrace();
166                    }
167            }
168    }
