421        /**
422         * Creates record (one or more, if not already there) with the specified files at the server side
423         * Notes that this method returns immediately after uploading the files to the server, (it blocks till then)
424         * the request is typically queued and generally served at a convenient time (for the server)
425         * @param indexFolder the root folder/file to index
426         * @param recursive whether to recursive descent into
427         * @param autoMerge whether to merge the records (per experiment) based on their dimensions
428         */
429        public List<Experiment> createRecord(File indexFolder, boolean recursive, boolean autoMerge)
430        {
431            try
432            {
433                    Class.forName("loci.formats.gui.BufferedImageReader");
434            }
435            catch(ClassNotFoundException notFound)
436            {
437                    throw new RuntimeException("bio-formats classes not found, please include loci_tools.jar in the classpath");
438            }
439    
440                    Indexer indexer = new Indexer();
441                    List<RawExperiment> exptList = indexer.findRecords(indexFolder, recursive, null); 
442                    if(exptList == null || exptList.isEmpty()) return null;
443                    
444                    List<RawExperiment> eList = new ArrayList<RawExperiment>();
445                    for(RawExperiment e : exptList)
446                    {
447                            if(e.isExistOnServer())
448                            {
449                                    System.out.println("Ignoring, Experiment already exist "+e);
450                            }
451                            else
452                            {
453                                    eList.add(e);
454                            }
455                    }
456                    
457                    System.out.println("Number of Experiments to upload :\t"+eList.size());
458                    if(autoMerge) 
459                    {
460                            for(RawExperiment expt : eList)
461                            {
462                                    mergeRecords(expt);
463                            }
464                    }
465                    
466                    List<Experiment> successList = new ArrayList<Experiment>();
467    
468                    for(RawExperiment expt : eList)
469                    {
470                            try
471                            {
472                                    uploadExperiment(expt);
473                                    successList.add( new Experiment(expt.getMD5Signature()) );
474                                    expt.dispose();
475                            } 
476                            catch (Exception e) 
477                            {
478                                    System.out.println("unable to upload records in experiment "+expt);
479                                    e.printStackTrace();
480                            }
481                    }
482                    
483                    return successList;
484        }

559            private void uploadExperiment(RawExperiment expt) throws IOException 
560            {
561                    Uploader uploader = getUploader(expt);
562                    System.out.println("zipping experiment source files "+expt);
563                    File tarBall = uploader.packSourceFiles();
564                    tarBall.deleteOnExit();
565                    System.out.println("created zipped file "+tarBall);
566                    
567                    try
568                    {
569                            //fetch a upload ticket first
570                            System.out.println("Fetching tickets for uploading experiment "+expt);
571                            Ticket ticket = uploader.fetchTicket();
572                            upload(ticket, tarBall);
573                    }
574                    finally
575                    {
576                            tarBall.delete();
577                    }
578            }
579            
580            private void upload(Ticket ticket, File tarBall) throws IOException
581            {
582                    URL uploadUrl = ticket.getUploadURL();
583                    HttpUtil httpUtil = new HttpUtil(uploadUrl);
584                    httpUtil.upload(tarBall, null);
585                    System.out.println("Successfully uploaded experiment archive "+tarBall);
586            }

588            private void mergeRecords(RawExperiment expt) 
589            {
590                    if(expt.size() == 1) return;// nothing to merge
591                    for(int i = expt.size(); i > 0 ; i--)
592                    {
593                            int counter = 0;
594                            for(Signature s : expt.getRecordSignatures())
595                            {
596                                    IRecord ithRecord = expt.getRecord(s);
597                                    counter++;
598                                    
599                                    if(ithRecord.getSiteCount() == 1)
600                                    {
601                                            List<IRecord> mergableRecords = expt.getAllSiblings(ithRecord);
602                                            if(mergableRecords.size() > 1)
603                                            {
604                                                    mergeRecords(expt, mergableRecords);
605                                                    break;
606                                            }
607                                    }
608                                    
609                                    if(counter == expt.size())
610                                            break;
611                            }
612                    }
613            }
614            
615            private void mergeRecords(RawExperiment expt, List<IRecord> mergableRecords)
616            {
617                    Signature[] signatureList = new Signature[mergableRecords.size()];
618                    for(int i = 0;i < signatureList.length; i++)
619                    {
620                            signatureList[i] = mergableRecords.get(i).getSignature();
621                    }
622                    
623                    expt.mergeRecordsAsSites(signatureList);
624            }
