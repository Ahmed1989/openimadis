021    package com.strandgenomics.imaging.iclient;
022    
023    import java.io.Serializable;
024    import java.util.ArrayList;
025    import java.util.List;
026    import java.util.Map;
027    import java.util.Set;
028    
029    import com.strandgenomics.imaging.icore.app.Parameter;
030    import com.strandgenomics.imaging.icore.app.Priority;
031    import com.strandgenomics.imaging.icore.util.Util;
032    
033    /**
034     * Represents an application to be executed (scheduled for execution) by the server
035     * @author arunabha
036     *
037     */
038    public class Application implements Serializable  {
039            
040            private static final long serialVersionUID = 5283925749816303859L;
041            /**
042             * name of the application
043             */
044            protected String name;
045            /**
046             * version of this application
047             */
048            public final String version; 
049            /**
050             * general description of the application
051             */
052            protected String description;
053            /**
054             * broad area (or domain) where this application is relevant - for example Quality Control, Statistics 
055             */
056            protected String categoryName;
057            /**
058             * name of the publisher
059             */
060            protected Publisher publisher = null;
061            /**
062             * list of parameters
063             */
064            protected Set<Parameter> parameters = null;
065            
066        Application(String name, String version, String category, String notes)
067            {
068                    this.name = name;
069                    this.version = version;
070                    this.categoryName = category;
071                    this.description = notes;
072            }
073    
074            public void dispose() 
075            {
076                    publisher = null;
077                    name = null;
078                    description = null;
079                    categoryName = null;
080            }
081            
082            /**
083             * Returns the publisher of this application
084             * @return
085             */
086            public synchronized Publisher getPublisher()
087            {
088                    if(publisher != null)
089                    {
090                            publisher = ImageSpaceObject.getImageSpace().getPublisher(this);
091                    }
092                    return publisher;
093            }
094            
095            /**
096             * Returns the name of the application
097             * @return the name of the application
098             */
099            public String getName()
100            {
101                    return name;
102            }
103            
104            /**
105             * Returns the version of this application
106             * @return the version of this application
107             */
108            public String getVersion()
109            {
110                    return version;
111            }
112            
113            /**
114             * Returns a short description of this application
115             * @return a short description of this application
116             */
117            public String getDescription()
118            {
119                    return description;
120            }
121            
122            /**
123             * Returns the category name 
124             * @return
125             */
126            public String getCategory()
127            {
128                    return this.categoryName;
129            }
130            
131            /**
132             * Returns a general description of each parameter needed for application execution
133             * @return a general description of each parameter needed for application execution
134             */
135            public synchronized Set<Parameter> getParameters()
136            {
137                    if(parameters == null)
138                    {
139                            parameters = ImageSpaceObject.getImageSpace().getParameters(this);
140                    }
141                    
142                    return parameters;
143            }
144            
145            /**
146             * execute the application on the specified records using the given parameters
147             * @param records the list of records to process
148             * @param parameters the parameter for the application execution
149             * @return the handle to the application-process for status etc
150             */
151            public Job execute(String projectName, List<Record> records, Map<String, Object> parameters, Priority priority)
152            {
153                    return ImageSpaceObject.getImageSpace().executeApplication(this, parameters, priority, projectName, getRecordGUIDs(records));
154            }
155            
156            /**
157             * execute the application on the specified records using the given parameters
158             * @param records the list of records to process
159             * @param parameters the parameter for the application execution
160             * @param priority the task priority
161             * @param scheduleTime the scheduled execution time
162             * @return the handle to the application-process for status etc
163             */
164            public Job schedule(String projectName, List<Record> records, Map<String, Object> parameters, Priority priority, long scheduleTime)
165            {
166                    return ImageSpaceObject.getImageSpace().scheduleApplication(this, parameters, priority, projectName, scheduleTime, getRecordGUIDs(records));
167            }
168            
169            /**
170             * Returns the list of available applications registered within the system
171             * @param categoryName name of the category or null
172             * @return the list of available applications registered within the system
173             */
174            public static List<Application> listApplications(String categoryName)
175            {
176                    return ImageSpaceObject.getImageSpace().listApplications(null, categoryName);
177            }
178            
179            private long[] getRecordGUIDs(List<Record> records)
180            {
181                    List<Long> guids = new ArrayList<Long>();
182                    for(Record r : records)
183                            guids.add( r.getGUID() );
184                    
185                    return Util.toLongArray(guids);
186            }
187    }
