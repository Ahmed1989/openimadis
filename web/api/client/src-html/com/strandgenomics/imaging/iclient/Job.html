021    package com.strandgenomics.imaging.iclient;
022    
023    import java.io.File;
024    import java.util.Map;
025    
026    import com.strandgenomics.imaging.icore.app.State;
027    
028    /**
029     * Represents a handle to a Task submitted to the server
030     * @author arunabha
031     *
032     */
033    public class Job extends ImageSpaceObject {
034            
035            private static final long serialVersionUID = 6750093791862505372L;
036            protected long id;
037            
038            Job(long id)
039            {
040                    this.id = id;
041            }
042            
043            public State getState()
044            {
045            //makes a system call
046            return getImageSpace().getJobState(this);
047            }
048            
049            /**
050             * returns the job identifier to identify the job at the server side
051             * @return
052             */
053            public long getJobID()
054            {
055                    return id;
056            }
057            
058            @Override
059            public void dispose() {
060                    // TODO Auto-generated method stub
061                    
062            }
063            
064        /**
065         * reschedules a task that has not yet commenced execution
066         */
067            public boolean reschedule(long rescheduleTime)
068            {
069                    return getImageSpace().rescheduleTask(this, rescheduleTime);
070            }
071    
072        /**
073         * pauses a task to be executed by the server
074         */
075            public boolean pause()
076            {
077                    return getImageSpace().pauseTask(this);
078            }
079            
080        /**
081         * resumes a task that has been paused
082         */
083            public boolean resume()
084            {
085                    return getImageSpace().resumeTask(this);
086            }
087            
088        /**
089         * deletes a task from the server task execution queue
090         */
091            public boolean remove()
092            {
093                    return getImageSpace().removeTask(this);
094            }
095            
096        /**
097         * Terminates an executing task
098         */
099            public boolean terminate()
100            {
101                    return getImageSpace().terminateTask(this);
102            }
103            
104            /**
105             * Returns the current progress of the task under execution
106             * @return the current progress of the task under execution
107             */
108            public int getProgress()
109            {
110                    return getImageSpace().getTaskProgress(this);
111            }
112            
113            /**
114             * sets the progress of the task under execution
115             * @param progress the current progress of the task under execution
116             */
117            public void setProgress(int progress)
118            {
119                    getImageSpace().setTaskProgress(id, progress);
120            }
121            
122            /**
123             * Returns the application parameters of this task
124             * @return  the application parameters of this task
125             */
126            public Map<String, Object> getParameters()
127            {
128                    return getImageSpace().getTaskParameters(this);
129            }
130            
131            /**
132             * Returns the guids of the input records of this task
133             * @return the guids of the input records of this task
134             */
135            public long[] getInputRecords()
136            {
137                    return getImageSpace().getTaskInputs(this);
138            }
139            
140            /**
141             * uploads the specified file as log of this task
142             * @param logFile
143             */
144            public void uploadLog(File logFile)
145            {
146                    getImageSpace().uploadTaskLog(this.id, logFile);
147            }
148    }
