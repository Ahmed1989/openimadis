022    package com.strandgenomics.imaging.icore.app;
023    
024    import java.util.Collection;
025    import java.util.HashSet;
026    import java.util.Set;
027    
028    /**
029     * The compute-server (worker-system) updates its status with the iManage server
030     * periodically with a request specified by this class
031     * @author arunabha
032     *
033     */
034    public class Request {
035            
036            /**
037             * state of the compute server (workers)
038             */
039            public final WorkerState state;
040            /**
041             * status of the list of jobs being executed
042             */
043            private final Set<JobReport> executingJobs;
044            
045            /**
046             * 
047             * @param state state of the compute worker
048             * @param jobs status of the list of the jobs
049             */
050            public Request(WorkerState state, Collection<JobReport> jobs)
051            {
052                    this.state = state;
053                    executingJobs = new HashSet<JobReport>();
054                    executingJobs.addAll(jobs);
055            }
056            
057            /**
058             * returns the jobs that are being reported by this request
059             * @return the jobs that are being reported by this request
060             */
061            public Collection<JobReport> getJobs()
062            {
063                    return executingJobs;
064            }
065    }
