071            /**
072             * Creates a RGB image from the rgb pixel values
073             * @param imageWidth the width of the image
074             * @param imageHeight the height of the image
075             * @param pixels the pixels value
076             * @return the Buffered Image created using the pixels
077             */
078            public static BufferedImage getRGBImage(int imageWidth, int imageHeight, int[] pixels)
079            {
080                    long startTime = System.currentTimeMillis();
081                    if(pixels.length != imageWidth * imageHeight)
082                            throw new IllegalArgumentException("pixels dimension mismatch");
083                    
084                    ColorModel cm = new DirectColorModel(32, 0x00FF0000, 0x0000FF00, 0x000000FF);
085                    
086                    WritableRaster wr = cm.createCompatibleWritableRaster(1, 1); //some dummy
087                    SampleModel sm = wr.getSampleModel();
088                    //create the relevant sample model
089                    sm = sm.createCompatibleSampleModel(imageWidth, imageHeight);
090                    //create the data buffer to hold the image
091                    DataBuffer db = new DataBufferInt(pixels, imageWidth*imageHeight, 0);
092                    //the relevant raster to write images
093                    WritableRaster r = Raster.createWritableRaster(sm, db, null);
094                    
095                    BufferedImage img = new BufferedImage(cm, r, false, null);
096                    
097                    long endTime = System.currentTimeMillis();
098                    logger.logp(Level.INFO, "PixelArray", "getRGBImage", "Created RGB Image in "+(endTime-startTime) +" ms");
099                    
100                    return img;
101            }

109            /**
110             * Overlay and return the RGB pixel values
111             * @param pixelDataList
112             * @return
113             */
114            public static Integer getRGBPixels(PixelArray ... pixelDataList)
115            {
116                    long startTime = System.currentTimeMillis();
117    
118                    ColorModel[] cms = new ColorModel[pixelDataList.length];
119                    byte[][] scaledPixels = new byte[pixelDataList.length][];
120                    
121                    for(int i = 0;i < pixelDataList.length; i++)
122                    {
123                            cms[i] = pixelDataList[i].getColorModel();
124                            scaledPixels[i] = pixelDataList[i].create8BitImage();
125                    }
126                    
127                    int imageWidth = pixelDataList[0].width;
128                    int imageHeight = pixelDataList[0].height;
129                    
130                    int noOfPixels = imageWidth * imageHeight;
131                    int[] overlayPixels = new int[noOfPixels];
132                    
133                    for(int i = 0;i < noOfPixels; i++)
134                    {
135                            int alpha = 255;
136                int red   = 0;
137                int green = 0;
138                int blue  = 0;
139    
140                            for(int j = 0;j < pixelDataList.length; j++)
141                            {
142                                    int pixel = scaledPixels[j][i];
143                                    
144                    red   += cms[j].getRed(pixel);
145                    green += cms[j].getGreen(pixel);
146                    blue  += cms[j].getBlue(pixel);
147    
148                    if (red > 255)   red   = 255;
149                    if (green > 255) green = 255;
150                    if (blue > 255)  blue  = 255;
151                            }
152                            
153                            overlayPixels[i] = ((alpha & 0xFF) << 24) |
154                            ((red & 0xFF) << 16)   |
155                            ((green & 0xFF) << 8)  |
156                            ((blue & 0xFF) << 0);
157                    }
158                    
159                    long endTime = System.currentTimeMillis();
160                    logger.logp(Level.INFO, "PixelArray", "getRGBPixels", "Overlaid "+ pixelDataList.length +" in "+(endTime-startTime) +" ms");
161                    return new Integer(overlayPixels, imageWidth, imageHeight);
162            }

164            /**
165         * Convert the buffer image to PixelArray 
166         * @param image a buffered image
167         * @return the pixel data of the buffered image
168         */
169        public static PixelArray toPixelArray(BufferedImage image)
170        {
171            if(image == null) return null;
172                    
173                    Raster raster = image.getData();
174                    DataBuffer data = raster.getDataBuffer();
175                    
176                    PixelArray pixels = null;
177                    
178                    if(data instanceof DataBufferByte)
179                    {
180                            pixels = new PixelArray.Byte(((DataBufferByte)data).getData(), image.getWidth(), image.getHeight());
181                    }
182                    else if(data instanceof DataBufferShort)
183                    {
184                            pixels = new PixelArray.Short(((DataBufferShort)data).getData(), image.getWidth(), image.getHeight());
185                    }
186                    else if(data instanceof DataBufferUShort)
187                    {
188                            pixels = new PixelArray.Short(((DataBufferUShort)data).getData(), image.getWidth(), image.getHeight());
189                    }
190                    else if(data instanceof DataBufferInt)
191                    {
192                            pixels = new PixelArray.Integer(((DataBufferInt)data).getData(), image.getWidth(), image.getHeight());
193                    }
194                    else if(data instanceof DataBufferFloat)
195                    {
196                            pixels = new PixelArray.Float(((DataBufferFloat)data).getData(), image.getWidth(), image.getHeight());
197                    }
198                    else
199                    {
200                            throw new IllegalStateException("unsupported buffered image "+data.getDataType());
201                    }
202                    
203                    return pixels;
204        }

206            /**
207             * Write this pixel array in the specified DataOutputStream
208             * @param out the output stream
209             * @throws IOException 
210             */
211            public static PixelArray read(InputStream in) throws IOException
212            {
213                    DataInputStream inStream = new DataInputStream(in);
214                    
215                    PixelDepth depth = PixelDepth.toPixelDepth( inStream.readInt() );
216                    int width  = inStream.readInt();
217                    int height = inStream.readInt();
218                    
219                    PixelArray rawData = null;
220                    
221                    switch(depth)
222                    {
223                            case INT:
224                            {
225                                    int[] pixels = new int[width*height];
226                                    for(int i = 0;i < pixels.length; i++)
227                                            pixels[i] = inStream.readInt();
228                                    
229                                    rawData = new Integer(pixels, width, height);
230                                    break;
231                            }
232                            case SHORT:
233                            {
234                                    short[] pixels = new short[width*height];
235                                    for(int i = 0;i < pixels.length; i++)
236                                            pixels[i] = inStream.readShort();
237                                    
238                                    rawData = new Short(pixels, width, height);
239                                    break;
240                            }
241                            case BYTE:
242                            {
243                                    byte[] pixels = new byte[width*height];
244                                    inStream.readFully(pixels);
245                                    
246                                    rawData = new Byte(pixels, width, height);
247                                    break;
248                            }               
249                    }
250                    
251                    return rawData;
252            }

385             * Returns the default grayscale IndexColorModel.
386         */
387            public synchronized IndexColorModel getDefaultColorModel() 
388            {
389                    if (defaultColorModel == null) 
390                    {
391                            byte[] r = new byte[256];
392                            byte[] g = new byte[256];
393                            byte[] b = new byte[256];
394                            
395                            for(int i=0; i<256; i++)
396                            {
397                                    r[i]=(byte)i;
398                                    g[i]=(byte)i;
399                                    b[i]=(byte)i;
400                            }
401                            IndexColorModel cm = new IndexColorModel(8, 256, r, g, b);
402                            defaultColorModel = cm;
403                    }
404                    return defaultColorModel;
405            }

407            /**
408             * Returns a look up table for gamma correction
409             * @param gamma a valid gamma value between 0.1 and 5.0
410             * @return the LUT
411             */
412            protected int[] getGammaLUT(double gamma)
413            {
414                    int[] lut = new int[256];
415                    int v = 0;
416                    
417                    for(int i=0; i<256; i++) 
418                    {
419                            v = (int)(Math.exp(Math.log(i/255.0)*gamma)*255.0);
420                            
421                            if (v < 0) v = 0;
422                            if (v > 255) v = 255;
423                            
424                            lut[i] = v;
425                    }
426                    
427                    return lut;
428            }

620         * The <code>Byte</code> class defines a PixelData specified
621         * in <code>byte</code> depth.
622         */
623        public static class Byte extends PixelArray implements Serializable
624        {
625                    private static final long serialVersionUID = -1347072227693291118L;
626                    private byte[] pixelArray;

637            {
638                            min = min & 0xFF;
639                            max = max & 0xFF;
640                            
641                    if(max <= min)
642                            throw new IllegalArgumentException("min("+min +") must be less then max ("+max+")");
643                    
644                    this.customMin = min;
645                    this.customMax = max;

662                }

685                    @Override
686                    public Histogram computeIntensityDistribution() 
687                    {
688                            //list of intensities with non zero frequencies
689                            OpenIntIntHashMap distribution = new OpenIntIntHashMap(255);
690    
691                            int maxFrequency = 0;
692                            minIntensity = 0xFF;
693                            maxIntensity = 0;
694                            
695                            for(int index = 0;index < pixelArray.length; index++)
696                            {
697                                    //the raw intensity value at index, last byte only
698                                    int intensityValue = pixelArray[index] & 0xFF;
699                                    if(intensityValue == 0) continue; //ignore zero intensity values
700                                    
701                                    int frequency = distribution.containsKey(intensityValue) ? distribution.get(intensityValue) : 0;
702                                    frequency += 1; //increment the pixel count at this intensity value
703                                    distribution.put(intensityValue, frequency);
704                                    
705                    if (maxFrequency < frequency) 
706                            maxFrequency = frequency;
707                    
708                    if(minIntensity > intensityValue)
709                            minIntensity = intensityValue;
710                    
711                    if(maxIntensity < intensityValue)
712                            maxIntensity = intensityValue;
713                            }
714                            
715                            if(minIntensity == 0xFF) minIntensity = 0; //if nothing is there
716                            distribution.trimToSize();
717                                    
718                            return new Histogram(getType(), distribution, minIntensity, maxIntensity, maxFrequency);
719                    }

721                    @Override
722                    public BigInteger computeMessageDigest(String algoName) throws NoSuchAlgorithmException 
723                    {
724                            MessageDigest md = MessageDigest.getInstance(algoName == null ? "MD5" : algoName);
725                            md.update(pixelArray);
726                            
727                            byte[] hash = md.digest();//128 bit number for MD5
728                            return new BigInteger(hash);
729                    }

733                            
734                            return new Byte(Arrays.copyOf(pixelArray, computePixelCount()), getWidth(), getHeight());
735                    }

737                    @Override
738                    public PixelArray create(int w, int h) 
739                    {
740                            Byte pArray = new Byte(new byte[w*h], w, h);
741                            
742                            ensureContrast(); //ensure contrast is set
743                            pArray.setContrast(this.getMinContrastSetting(), this.getMaxContrastSetting());
744                            pArray.setGamma(this.getGamma());
745                            pArray.setColorModel(this.getColorModel());
746                            
747                            return pArray;
748                    }

750                    @Override
751                    public PixelArray getSubArray(int x, int y, int w, int h) 
752                    {
753                            if(x < 0 || y < 0 || w <= 0 || h <= 0 ||
754                                            (x+w) > this.width || (y+h) > this.height)
755                                    throw new IllegalArgumentException("specified area is not contained within this PixelArray");
756                            
757                            Byte subArray = (Byte) create(w, h);
758                            byte[] bachha = subArray.pixelArray;
759                            
760                            for (int j = y; j < y+h; j++) 
761                            {
762                                    int offset1 = (j - y) * w;
763                                    int offset2 = j * width + x;
764                                    
765                                    for (int i = 0; i < w; i++)
766                                            bachha[offset1++] = pixelArray[offset2++];
767                            }
768    
769                            return subArray;
770                    }

772                    @Override
773                    public void overlay(PixelArray another) 
774                    {
775                            if(!(another instanceof Byte))
776                                    throw new IllegalArgumentException("overlay can be done with the same type");
777                            
778                            byte[] otherArray = ((Byte)another).pixelArray;
779    
780                            for (int i = 0; i < pixelArray.length; i++)
781                            {
782                                    // to handle unsigned case
783                                    int x = pixelArray[i] & 0xFF;
784                                    int y = otherArray[i] & 0xFF;
785                                    
786                                    if (x < y)
787                                    {
788                                            pixelArray[i] = otherArray[i];
789                                    }
790                            }
791                    }
