020    package com.strandgenomics.imaging.icore;
021    
022    import java.io.Serializable;
023    
024    /**
025     * Represents a channel of a record
026     * @author arunabha
027     */
028    public class Channel implements IChannel, Serializable {
029            
030            private static final long serialVersionUID = -2187489346863510931L;
031            /**
032             * name of the channel
033             */
034            protected String name;
035            /**
036             * wavelength of the channel light source (in micrometre)
037             */
038            protected int wavelength = -1;
039            /**
040             * color associated with this channel, default is green
041             */
042            protected int rgbColor = 0xFF00FF00;
043            
044            /**
045             * lut associated with channel
046             */
047            protected String lut = "grays";
048            /**
049             * user can set a custom contrast for a channel in a record
050             */
051            protected VisualContrast customContrast = null;
052            /**
053             * contrast for the z-stacked image
054             */
055            protected VisualContrast zstackContrast = null;
056            
057            /**
058             * Revision number for channel
059             * Any change to lut, contrast will increment the revision number 
060             */
061            protected long revision=0;
062            
063            /**
064             * creates a default channel 
065             * @param name name of this channel
066             */
067            public Channel(String name)
068            {
069                    this(name, 0xFF00FF00);
070            }
071            
072            /**
073             * Creates a channel with the specified name, color and contrast
074             * @param name name of the channel
075             * @param color color associated with this channel
076             */
077            public Channel(String name, int color)
078            {
079                    this.name = name;
080                    this.rgbColor = color;
081                    
082            }
083            
084            /**
085             * Creates a channel with the specified name and lut
086             * @param name name name of the channel
087             * @param lut name of the lut
088             */
089            public Channel(String name, String lut)
090            {
091                    this.name = name;
092                    this.lut = lut;
093            }
094            
095            /**
096             * Returns the name of this channel
097             * @return the name of this channel
098             */
099            public String getName()
100            {
101                    return this.name;
102            }
103            
104            public void setName(String value)
105            {
106                    this.name = value;
107            }
108            
109            /**
110             * Returns the color associated with this channel
111             * @return the color associated with this channel
112             */
113            public int getColor()
114            {
115                    return this.rgbColor;
116            }
117            
118            /**
119             * Returns the lut associated with this channel
120             * @return the lut associated with this channel
121             */
122            public String getLut()
123            {
124                    return this.lut == null ? "grays" : this.lut;
125            }
126            
127            /**
128             * Sets the lut associated with this channel
129             * @param lut the name of lut to be associated with this channel
130             */
131            public void setLut(String lut)
132            {
133                    this.lut = lut;
134            }
135            
136        /**
137         * Sets a custom contrast for all images of this channel of the owning record
138         * @param zStacked true if zStacked contrast, false otherwise
139         * @param contrast the custom contrast to set
140         */
141        public void setContrast(boolean zStacked, VisualContrast contrast)
142        {
143            if(zStacked)
144                    this.zstackContrast = contrast;
145            else
146                    this.customContrast = contrast;
147        }
148        
149        /**
150         * Returns the custom contrast, if any, for all images of this channel of the owning record
151         * @param zStacked true if zStacked contrast, false otherwise
152         * @return the custom contrast, if any, for all images of this channel of the owning record, otherwise null
153         */
154        public VisualContrast getContrast(boolean zStacked)
155        {
156            if(zStacked)
157                    return this.zstackContrast;
158            
159            return customContrast;
160        }
161        
162        public int getWavelength()
163        {
164            return wavelength;
165        }
166        
167        
168        public void setWavelength(int waveLength)
169        {
170            this.wavelength = waveLength;
171        }
172        
173        
174        public long getRevision() {
175                    return revision;
176            }
177        
178        
179        public void setRevision(long revision) {
180                    this.revision = revision;
181            }
182        
183        
184        public String toString()
185        {
186            if(wavelength > 0)
187            {
188                    return name +"("+wavelength+")";
189            }
190            else
191            {
192                    return name;
193            }
194        }
195        
196        @Override
197        public int hashCode()
198        {
199            return this.name.hashCode();
200        }   
201        
202        @Override
203            public boolean equals(Object obj)
204            {
205            if(obj != null && obj instanceof Channel)
206                    {
207                    Channel that = (Channel) obj;
208                            if(this == that) return true;
209                            
210                            boolean equals = this.lut.equals(that.lut) && this.name.equals(that.name);
211                            if (this.customContrast != null)
212                                equals = equals && this.customContrast.equals(that.customContrast);
213                            if (this.zstackContrast != null)
214                                equals = equals && this.zstackContrast.equals(that.zstackContrast);
215                            return equals;
216                    }
217                    return false;
218            }
219    }
