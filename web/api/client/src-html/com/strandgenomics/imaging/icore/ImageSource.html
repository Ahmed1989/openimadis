020    package com.strandgenomics.imaging.icore;
021    
022    import java.awt.image.BufferedImage;
023    import java.io.IOException;
024    
025    /**
026     * A source or provider of image within a record
027     * @author arunabha
028     *
029     */
030    public interface ImageSource {
031            
032        /**
033         * Returns the width (in pixels) of the images represented by this source
034         * Note that width is same for all images within the record
035         */
036        public int getImageWidth();
037        
038        /**
039         * Returns the height (in pixels) of the images represented by this source
040         * Note that height is same for all images within the record
041         */
042        public int getImageHeight();
043        
044            /**
045             * Returns the raw BufferedImage instance associated with this image source
046             * @param useChannelColor if true a color image (based on the channel color) is returned, gray-scale image otherwise
047             * @return BufferedImage instance associated with this image source
048             */
049            public BufferedImage getImage(boolean useChannelColor) throws IOException;
050    }
