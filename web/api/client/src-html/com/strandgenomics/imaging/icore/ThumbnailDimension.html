001    package com.strandgenomics.imaging.icore;
002    
003    /**
004     * Dimensions that identify a configuration of image(s) and used to generate thumbnail
005     * 
006     * @author Anup Kulkarni
007     */
008    public class ThumbnailDimension extends VODimension {
009    
010            /**
011             * list of channels
012             */
013            private int[] channelNos;
014            /**
015             * true if channels are tiled; false if overlayed
016             */
017            private boolean mosaic;
018            /**
019             * true if images are overlayed across slices; false if single slice
020             */
021            private boolean zStack;
022            /**
023             * true if channel colors are used; false in case of gray scale
024             */
025            private boolean useChannelColor;
026    
027            public ThumbnailDimension(int frameNumber, int sliceNumber, int siteNo, int[] channelNos, boolean mosaic, boolean zStack, boolean useChannelColor) {
028                    super(frameNumber, sliceNumber, siteNo);
029                    
030                    this.channelNos = channelNos;
031                    this.mosaic = mosaic;
032                    this.zStack = zStack;
033                    this.useChannelColor = useChannelColor;
034            }
035    
036            public int[] getChannelNos()
037            {
038                    return channelNos;
039            }
040    
041            public boolean isMosaic()
042            {
043                    return mosaic;
044            }
045    
046            public boolean isZStack()
047            {
048                    return zStack;
049            }
050    
051            public boolean isUseChannelColor()
052            {
053                    return useChannelColor;
054            }
055    
056    }
