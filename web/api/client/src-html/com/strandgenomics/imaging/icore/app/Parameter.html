020    package com.strandgenomics.imaging.icore.app;
021    
022    import java.io.Serializable;
023    
024    /**
025     * Description of a parameter of an application
026     * @author arunabha
027     *
028     */
029    public class Parameter implements Serializable {
030    
031            private static final long serialVersionUID = -4442491119645663893L;
032            
033            /**
034             * name of the parameter
035             */
036            public final String name;
037            /**
038             * type of the parameter, one of integer, float, boolean and string only
039             */
040            public final ParameterType type;
041            /**
042             * constraints on this parameter, can be null, meaning unbounded
043             */
044            public final ParameterConstraints constraints;
045            /**
046             * default value associated with this parameter, null is not a valid value
047             */
048            public final Object defaultValue;
049            /**
050             * description of the parameter
051             */
052            public final String description;
053            
054            /**
055             * Creates a Parameter having the specified name, type and constraints
056             * @param name name of the parameter
057             * @param type type of the parameter, one of integer, float boolean and string only
058             * @param constraints constraints on this parameter
059             * @param defaultValue default value of the parameter
060             * @param description parameter description
061             */
062            public Parameter(String name, ParameterType type, ParameterConstraints constraints, Object defaultValue, String description)
063            {
064                    if(name == null || type == null)
065                            throw new IllegalArgumentException("null parameter name or type");
066                    
067                    if(constraints != null && constraints.getType() != type)
068                            throw new IllegalArgumentException("parameter and its constraint type do not match");
069                    
070                    this.name = name;
071                    this.type = type;
072                    this.constraints = constraints;
073                    
074                    if(!validate(defaultValue))
075                            throw new IllegalArgumentException("illegal default value="+defaultValue+" for parameter "+name);
076                    
077                    this.defaultValue = defaultValue;
078                    
079                    this.description = description;
080            }
081            
082            @Override
083            public String toString()
084            {
085                    return name;
086            }
087            
088            @Override
089            public int hashCode()
090            {
091                    return name.hashCode();
092            }
093            
094            @Override
095            public boolean equals(Object obj)
096            {
097                    if(obj != null && obj instanceof Parameter)
098                    {
099                            Parameter that = (Parameter) obj;
100                            if(this == that) 
101                                    return true;
102                            else 
103                                    return this.name.equals(that.name);
104                    }
105                    return false;
106            }
107            
108            /**
109             * check if the specified value is valid for this parameter
110             * @param value the value to verify
111             * @return true if valid, false otherwise
112             */
113            public boolean validate(Object value)
114            {
115                    if(isValidType(value))
116                    {
117                            return constraints == null ? true : constraints.validate(value); 
118                    }
119                    else
120                    {
121                            return false;
122                    }
123            }
124            
125            /**
126             * Returns true if the specified value is consistent with this parameter's type
127             * @param value the value to verify
128             * @return  true if valid, false otherwise
129             */
130            public boolean isValidType(Object value)
131            {
132                    boolean validType = false;
133                    switch(type)
134                    {
135                            case INTEGER:
136                                    validType = (value instanceof Long || value instanceof Integer || value instanceof Short);
137                                    break;
138                            case DECIMAL:
139                                    validType = (value instanceof Float || value instanceof Double);
140                                    break;
141                            case STRING:
142                                    validType = (value instanceof String);
143                                    break;
144                            case BOOLEAN:
145                                    validType = (value instanceof Boolean);
146                                    break;
147                    }
148                    return validType;
149            }
150    }
