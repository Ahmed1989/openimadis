021    package com.strandgenomics.imaging.icore.app;
022    
023    /**
024     * The response to a ping request received by the compute server (worker-system) 
025     * @author arunabha
026     *
027     */
028    public class Response {
029            
030            /**
031             * the iManage server command to the compute server (worker-system)
032             */
033            public final Directive command; 
034            /**
035             * the job to execute, if any, null otherwise
036             */
037            public final Work job;
038            
039            /**
040             * Construct a iManage server response with only a directive
041             * @param command the directive to the compute-server
042             */
043            public Response(Directive command)
044            {
045                    this.command = command;
046                    this.job = null;
047            }
048            
049            /**
050             * Construct a iManage server response with the job and the associated directive
051             * @param command the directive to the compute-server
052             * @param task the task to execute/terminate/etc by the compute server 
053             */
054            public Response(Directive command, Work task)
055            {
056                    this.command = command;
057                    this.job = task;
058            }
059    }
