020    package com.strandgenomics.imaging.icore;
021    
022    /**
023     * An pixel-data (raw bitmap image) within a record is uniquely identified by the 4 dimensions 
024     * - (slices, frames, sites and channels). 
025     * This are in addition to its x and y coordinates which is fixed for all pixel-data (images) within a record. 
026     * All this 4 numbers starts with 0 and the last value is limited by the size in each corresponding dimensions
027     * 
028     * Dimension class represents the four dimension of a pixel-data that is used to identify it within a given record 
029     * (excluding the x-y dimension)
030     * 
031     * @author arunabha
032     */
033    public class Dimension extends VODimension implements Comparable<Dimension> {
034            
035            private static final long serialVersionUID = -5219011958689941797L;
036    
037            /**
038             * the channel number
039             */
040            public final int channelNo;
041    
042            /**
043             * Creates a Dimension instance with the specified frame,slice,site,channel numbers
044             * @param frameNumber the frame number for a pixel-data with a record
045             * @param sliceNumber the slice number for a pixel-data with a record
046             * @param channelNumber the channel number for a pixel-data with a record
047             */
048            public Dimension(int frameNumber, int sliceNumber, int channelNumber, int siteNumber)
049            {
050                    super(frameNumber, sliceNumber, siteNumber);
051                    this.channelNo = channelNumber;
052            }
053    
054            @Override
055            public int hashCode()
056            {
057                    return (((frameNo & 0xFF) << 24) + ((sliceNo & 0xFF) << 16) + ((siteNo & 0xFF) << 8) + ((channelNo & 0xFF) << 0));
058            }
059            
060            @Override
061            public boolean equals(Object obj)
062            {
063                    if(obj != null && obj instanceof Dimension)
064                    {
065                            Dimension that = (Dimension) obj;
066                            return (this.frameNo == that.frameNo &&
067                                            this.sliceNo == that.sliceNo &&
068                                            this.channelNo == that.channelNo &&
069                                            this.siteNo == that.siteNo);
070                    }
071                    return false;
072            }
073            
074            @Override
075            public String toString()
076            {
077                    StringBuilder builder = new StringBuilder();
078                    builder.append('{');
079                    builder.append(this.frameNo).append(',');
080                    builder.append(this.sliceNo).append(',');
081                    builder.append(this.channelNo).append(',');
082                    builder.append(this.siteNo);
083                    builder.append('}');
084                    
085                    return builder.toString();
086            }
087    
088            /**
089             * the order is frame slice,channel and then site
090             */
091            @Override
092            public int compareTo(Dimension that)
093            {
094                    //returns a negative integer, zero, or a positive integer as this object is less than, 
095                    //equal to, or greater than the specified object. 
096                    if(this.frameNo < that.frameNo)
097                            return -1;
098                    else if(this.frameNo > that.frameNo)
099                            return 1;
100                    else 
101                    {
102                            if(this.sliceNo < that.sliceNo)
103                                    return -1;
104                            else if(this.sliceNo > that.sliceNo)
105                                    return 1;
106                            else
107                            {
108                                    if(this.channelNo < that.channelNo)
109                                            return -1;
110                                    else if(this.channelNo > that.channelNo)
111                                            return 1;
112                                    else
113                                    {
114                                            if(this.siteNo < that.siteNo)
115                                                    return -1;
116                                            else if(this.siteNo > that.siteNo)
117                                                    return 1;
118                                            else
119                                                    return 0;
120                                    }
121                            }
122                    }
123            }
124    }
