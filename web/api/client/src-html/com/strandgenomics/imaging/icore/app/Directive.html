022    /**
023     * The directive received by a compute-worker-system from the iManage server
024     * in response to a hand-shake established between the worker and the server
025     * @author arunabha
026     *
027     */
028    public enum Directive {
029    
030        /**
031         * an unknown worker-request has been received
032         */
033        UNKNOWN_REQUEST,
034        /**
035         * the worker is free & no pending task for him
036         */
037        WAIT,
038        /**
039         * the worker is busy with some task, let him do that
040         */
041        CONTINUE,
042        /**
043         * unknown worker, he must register himself first
044         */
045        REGISTRATION_REQUIRED,
046        /**
047         * request to workers to publish its applications
048         */
049        PUBLISH_APPLICATIONS,
050        /**
051         * the worker is free, he gets a task to execute
052         */
053        EXECUTE_TASK,
054        /**
055         * request to kill the task, the worker is executing
056         */
057        TERMINATE_TASK,
058        /**
059         * terminate the worker
060         */
061        TERMINATE;
062    }
