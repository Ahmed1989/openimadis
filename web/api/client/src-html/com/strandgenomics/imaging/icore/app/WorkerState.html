022    /**
023     * The state of a compute-server system
024     * @author arunabha
025     *
026     */
027    public enum WorkerState {
028    
029            /**
030             * the compute-server system is fully free and waiting to get a task; also
031             * compute-sever system has nothing to report anything about previsous
032             * requests
033             */
034            FREE,
035            /**
036             * the compute-server system is partially busy, serving one or more tasks,
037             * but can still pick up more, the compute-server system may be completely
038             * free in sense of not having any task scheduled to run, but still have
039             * some task reports to submit about previously allocated tasks (eg.
040             * SUCCESSFUL, TERMINATED, FALIED etc)
041             */
042            ENGAGED,
043            /**
044             * the compute-server system is fully busy, serving one or more tasks
045             */
046            BUSY
047    }
