020    package com.strandgenomics.imaging.icore;
021    
022    /**
023     * Represents a channel of a record
024     * @author arunabha
025     */
026    public interface IChannel {
027            
028            /**
029             * Returns the name of this channel
030             * @return the name of this channel
031             */
032            public String getName();
033            
034            /**
035             * Returns the wavelength of the channel light source
036             * @return wavelength in micrometre if set, else -1
037             */
038            public int getWavelength();
039            
040            /**
041             * Returns the lut name associated with this channel
042             * @return the lut name associated with this channel
043             */
044            public String getLut();
045            
046        /**
047         * Returns the custom contrast, if any, for all images of this channel of the owning record
048         * @param zStacked true if zStacked contrast, false otherwise
049         * @return the custom contrast, if any, for all images of this channel of the owning record, otherwise null
050         */
051        public VisualContrast getContrast(boolean zStacked);
052    }
