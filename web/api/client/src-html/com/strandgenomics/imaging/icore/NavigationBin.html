020    package com.strandgenomics.imaging.icore;
021    
022    import java.sql.Timestamp;
023    import java.util.HashSet;
024    import java.util.Set;
025    
026    
027    /**
028     * Represents a new FilterCondition on top of an existing list of FilterConditions 
029     * @author arunabha
030     */
031    public class NavigationBin extends SearchCondition {
032            
033            private static final long serialVersionUID = 6290841382726390110L;
034            /**
035             * list of pre-existing filter conditions
036             */
037            protected Set<SearchCondition> preExistingConditions = null;
038            /**
039             * number of records satisfying the filters
040             */
041            protected int noOfRecords = 0;
042            
043            /**
044             * Creates a bin with the specified filters and number of records satisfying the filters
045             * @param filters list of pre-existing filter conditions, can be null
046             * @param fieldName name of the field to search on
047             * @param lowerLimit lower limit of the value
048             * @param upperLimit upper limit of the value
049             */
050            public NavigationBin(Set<SearchCondition> filters, String fieldName, long lowerLimit, long upperLimit)
051            {
052                    super(fieldName, lowerLimit, upperLimit);
053                    this.preExistingConditions = filters;
054            }
055            
056            /**
057             * Creates a bin with the specified filters and number of records satisfying the filters
058             * @param filters list of pre-existing filter conditions, can be null
059             * @param fieldName name of the field to search on
060             * @param lowerLimit lower limit of the value
061             * @param upperLimit upper limit of the value
062             */
063            public NavigationBin(Set<SearchCondition> filters, String fieldName, double lowerLimit, double upperLimit)
064            {
065                    super(fieldName, lowerLimit, upperLimit);
066                    this.preExistingConditions = filters;
067            }
068            
069            /**
070             * Creates a bin with the specified filters and number of records satisfying the filters
071             * @param filters list of pre-existing filter conditions, can be null
072             * @param fieldName name of the field to search on
073             * @param lowerLimit lower limit of the value
074             * @param upperLimit upper limit of the value
075             */
076            public NavigationBin(Set<SearchCondition> filters, String fieldName, String lowerLimit, String upperLimit)
077            {
078                    super(fieldName, lowerLimit, upperLimit);
079                    this.preExistingConditions = filters;
080            }
081            
082            /**
083             * Creates a bin with the specified filters and number of records satisfying the filters
084             * @param filters list of pre-existing filter conditions, can be null
085             * @param fieldName name of the field to search on
086             * @param lowerLimit lower limit of the value
087             * @param upperLimit upper limit of the value
088             */
089            public NavigationBin(Set<SearchCondition> filters, String fieldName, Timestamp lowerLimit, Timestamp upperLimit)
090            {
091                    super(fieldName, lowerLimit, upperLimit);
092                    this.preExistingConditions = filters;
093            }
094            
095            /**
096             * Creates an empty bin
097             * @param filters list of pre-existing filter conditions, can be null
098             * @param field the search field
099             */
100            public NavigationBin(Set<SearchCondition> filters, SearchField field)
101            {
102                    super(field.fieldName, field.fieldType);
103                    this.preExistingConditions = filters;
104            }
105            
106            public Set<SearchCondition> getPreExitingFilters()
107            {
108                    return preExistingConditions;
109            }
110            
111            public void setRecordCount(int count)
112            {
113                    noOfRecords = count;
114            }
115            
116            public int getRecordCount()
117            {
118                    return noOfRecords;
119            }
120            
121            public Set<SearchCondition> toFilter()
122            {
123                    Set<SearchCondition> filters = new HashSet<SearchCondition>();
124                    if(preExistingConditions != null) filters.addAll(preExistingConditions);
125                    filters.add( this );
126                    return filters;
127            }
128    }
