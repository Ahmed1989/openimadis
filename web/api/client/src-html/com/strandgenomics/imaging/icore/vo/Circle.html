020    package com.strandgenomics.imaging.icore.vo;
021    
022    import java.awt.geom.Rectangle2D;
023    
024    /**
025     * The <code>Circle</code> class describes an circle that is defined by a framing rectangle.
026     * @author navneet
027     */
028    public class Circle extends VisualObject {
029    
030            private static final long serialVersionUID = 5399819711012539649L;
031            /**
032             * the bounding rectangle
033             */
034            protected Rectangle2D.Double boundingRect = new Rectangle2D.Double();
035        
036        /**
037         * Creates an Circle instance from the specified coordinates.
038         *
039         * @param x the X coordinate of the upper-left corner of the framing rectangle
040         * @param y the Y coordinate of the upper-left corner of the framing rectangle
041         * @param w the width of the framing rectangle
042         * @param h the height of the framing rectangle
043         */
044        public Circle(double x, double y, double width, double height) 
045        {
046            super(VisualObjectType.CIRCLE);
047            setFrame(x, y, width, height);
048        }
049        
050        /**
051         * Creates an Circle instance from the specified coordinates.
052         *
053         * @param x the X coordinate of the upper-left corner of the framing rectangle
054         * @param y the Y coordinate of the upper-left corner of the framing rectangle
055         * @param w the width of the framing rectangle
056         * @param h the height of the framing rectangle
057         */
058        public Circle(int ID, double x, double y, double width, double height) 
059        {
060            super(VisualObjectType.CIRCLE, ID);
061            setFrame(x, y, width, height);
062        }
063        
064        /**
065         * updates this Circle with the specified coordinates.
066         *
067         * @param x the X coordinate of the upper-left corner of the framing rectangle
068         * @param y the Y coordinate of the upper-left corner of the framing rectangle
069         * @param w the width of the framing rectangle
070         * @param h the height of the framing rectangle
071         */
072        public void setFrame(double x, double y, double width, double height) 
073        {
074                    boundingRect.x = x;
075                    boundingRect.y = y;
076                    boundingRect.width = width;
077                    boundingRect.height = height;
078        }
079    
080            @Override
081            public Rectangle2D.Double getBounds()
082            {
083                    return boundingRect;
084            }
085    }
