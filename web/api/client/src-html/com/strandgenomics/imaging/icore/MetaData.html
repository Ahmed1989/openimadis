020    package com.strandgenomics.imaging.icore;
021    
022    import java.math.BigDecimal;
023    import java.math.BigInteger;
024    import java.sql.Timestamp;
025    import java.text.DateFormat;
026    import java.util.Calendar;
027    import java.util.Date;
028    
029    import com.strandgenomics.imaging.icore.util.Util;
030    
031    /**
032     * 
033     * metadata represented as Name-Value pair
034     * @author arunabha
035     */
036    public class MetaData implements Storable {
037    
038            private static final long serialVersionUID = 465002750937072912L;
039            
040            /**
041         * Formatter to use for date
042         */
043        private static final DateFormat dateFormat = DateFormat.getDateTimeInstance(DateFormat.SHORT, DateFormat.SHORT);
044        
045            /**
046             * name of the meta data
047             */
048            protected String name;
049            /**
050             * the user annotations value
051             */
052            protected Object value;
053            /**
054             * type of this annotation
055             */
056            protected AnnotationType type;
057            
058            public MetaData(String name, Long value)
059            {
060                    this.name = name;
061                    this.value = value;
062                    this.type = AnnotationType.Integer;
063            }
064            
065            public MetaData(String name, String value) 
066            {
067                    this.name = name;
068                    this.value = value;
069                    this.type = AnnotationType.Text;
070            }
071            
072            public MetaData(String name, Double value) 
073            {
074                    this.name = name;
075                    this.value = value;
076                    this.type = AnnotationType.Real;
077            }
078            
079            public MetaData(String name, Timestamp value) 
080            {
081                    this.name = name;
082                    this.value = value;
083                    this.type = AnnotationType.Time;
084            }
085            
086            public static MetaData createInstance(String name, Object value)
087            {
088            if(value == null)
089                return null;
090    
091            if(value instanceof String)
092            {
093                return new MetaData(name, (String)value);
094            }
095            else if(value instanceof Double || value instanceof Float || value instanceof BigDecimal)
096            {
097                    return new MetaData(name, Util.getDouble(value));      
098            }
099            else if(value instanceof Long || value instanceof Integer || 
100                            value instanceof BigInteger || value instanceof Short || value instanceof Byte){
101                        
102                    return new MetaData(name, Util.getLong(value));      
103            }
104            else if(value instanceof Calendar)
105            {
106                    return new MetaData(name, new Timestamp(((Calendar) value).getTimeInMillis()));      
107            }
108            else if(value instanceof Date)
109            {
110                    return new MetaData(name, new Timestamp(((Date)value).getTime()));      
111            }
112            else if(value instanceof java.sql.Date)
113            {
114                    return new MetaData(name, new Timestamp(((java.sql.Date)value).getTime()));      
115            }
116            else if(value instanceof Timestamp)
117            {
118                    return new MetaData(name, (Timestamp)value);      
119            }
120            else
121            {
122                    throw new IllegalArgumentException("unknown annotation value type");
123            }
124            }
125            
126            public AnnotationType getType()
127            {
128                    return this.type;
129            }
130    
131            /**
132             * @return the name
133             */
134            public String getName() 
135            {
136                    return name;
137            }
138    
139            /**
140             * @param name the name to set
141             */
142            public void setName(String name) 
143            {
144                    this.name = name;
145            }
146            
147            public Object getValue()
148            {
149                    return value;
150            }
151    
152            /**
153             * @return get string value of the metadata
154             */
155            public String stringValue() {
156                switch (type) {
157                case Integer:
158                case Real:
159                case Text:
160                    return value+"";
161                case Time:
162                    return dateFormat.format(value);
163                }
164                return super.toString();
165            }
166    
167        @Override
168        public String toString() {
169            return "[" + this.name + "," + this.type + "," + this.value + "]";
170        }
171    
172            @Override
173            public void dispose() 
174            {}
175    }
