020    package com.strandgenomics.imaging.icore.app;
021    
022    public enum ParameterType {
023            
024            BOOLEAN(Boolean.class, "boolean"),
025            INTEGER(Integer.class, "int"),
026            DECIMAL(Double.class, "double"),
027            STRING(String.class, "String");
028            
029            private final String description;
030            private final Class<?> clazz;
031            
032            private ParameterType(Class<?> clazz, String description)
033            {
034                    this.description = description;
035                    this.clazz = clazz;
036            }
037            
038            @Override
039            public String toString()
040            {
041                    return description;
042            }
043            
044            public boolean isValidValue(Object value)
045            {
046                    return value == null || clazz.isInstance(value);
047            }
048    }
