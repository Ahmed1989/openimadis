020    package com.strandgenomics.imaging.icore;
021    
022    /**
023     * Specification of a source file producing an experiment (muti-series records)
024     * @author arunabha
025     *
026     */
027    public interface ISourceReference {
028            
029            /**
030             * Returns the actual name of the source file (fully qualified absolute path in the acquisition machine)
031             * @return name of the source file
032             */
033            public String getSourceFile();
034            
035            /**
036             * Returns the size of the source file
037             * @return the size of the source file
038             */
039            public long getSize();
040            
041            /**
042             * Returns the time when this source was last modified
043             * @return the time when this source was last modified
044             */
045            public long getLastModified();
046    }
